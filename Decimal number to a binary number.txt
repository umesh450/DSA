
#include <stdio.h>  

#define SIZE 50      
int stack[SIZE];     
int top = -1;        
void push(int value) {
    if (top == SIZE - 1) {        
        printf("Stack Overflow! Cannot push %d\n", value);
    } else {
        top++;                    
        stack[top] = value;        
    }
}

int pop() {
    if (top == -1) {              
        printf("Stack Underflow!\n");
        return -1;                
    } else {
        int value = stack[top];   
        top--;                     
        return value;           
    }
}

int main() {
    int decimal, remainder;   

   
    printf("Enter a decimal number: ");
    scanf("%d", &decimal);

   
    if (decimal == 0) {
        printf("Binary: 0\n");
        return 0;
    }

   
    while (decimal > 0) {
        remainder = decimal % 2;  
        push(remainder);          
        decimal = decimal / 2;    
    }

   
    printf("Binary: ");
    while (top != -1) {            
        printf("%d", pop());      
    }
    printf("\n");

    return 0;  // End of program
}
